class Solution {
   int linearSearch(int nums[],int target){
    int n=nums.size();
    for(int i = 0; i < n; i++) {
        if (nums[i] == target) return i;
    }
    return -1;
    }
};
Time Complexity 
                             1.Best-case: O(1) 
2.Worst-case: O(n) 
3.Average-case: O(n)
